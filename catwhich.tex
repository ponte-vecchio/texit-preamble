\def\catwhich#1{%
	\def\curcat{\the\catcode`#1}%
	\edef\catdesc{\matchcase{\curcat}{%
		{0}{Escape~Character}
		{1}{Begin~Character}
		{2}{End~Group~Character}
		{3}{Math~Shift~Character}
		{4}{Alignment~Tab}
		{5}{EOL}
		{6}{Parameter}
		{7}{Superscript}
		{8}{Subscript}
		{9}{Ignored~Character}
		{10}{Space}
		{11}{Letter}
		{12}{Other}
		{13}{Active~Character}
		{14}{Comment~Character}
		{15}{Invalid~Character}
	}}
	{\sf\halign{
		\hfil\small##\hfil\cr
		\textcolor{brightyellow}{{\escapechar-1 \string#1}}\quad$\longrightarrow$\quad\textcolor{brightmagenta}{\bfseries\curcat}\cr
		\hfil\small\textcolor{brightblue}{\slshape\catdesc}\hfil\cr
	}}
}
